408 AddLogLineCS(wxT("Failed to open 'ED2KFile', cannot signal running instance."))theLogger.AddLogLine(L"amuleAppCommon.cpp", 408, true, logStandard
, L"Failed to open 'ED2KFile', cannot signal running instance."
, true);
409 }
410
411 return false;
412 } else {
413 AddLogLineNS(wxT("No other instances are running."))theLogger.AddLogLine(L"amuleAppCommon.cpp", 413, false, logStandard
, L"No other instances are running.", true);
414 }
415#endif
416
417#ifndef __WINDOWS__
418 // Close standard-input
419 if ( !cmdline.Found(wxT("enable-stdin")L"enable-stdin") ) {
420 // The full daemon will close all std file-descriptors by itself,
421 // so closing it here would lead to the closing on the first open
422 // file, which is the logfile opened below
423 if (!enable_daemon_fork) {
424 close(0);
425 }

428
429 // Create the CFG file we shall use and set the config object as the global cfg file
430 wxConfigwxFileConfig::Set(new wxFileConfig( wxEmptyString, wxEmptyString, thePrefs::GetConfigDir() + m_configFile));
431
432 // Make a backup of the log file
433 CPath logfileName = CPath(thePrefs::GetConfigDir() + m_logFile);
434 if (logfileName.FileExists()) {
435 CPath::BackupFile(logfileName, wxT(".bak")L".bak");
436 }
437
438 // Open the log file
439 if (!theLogger.OpenLogfile(logfileName.GetRaw())) {
440 // use std err as last resolt to indicate problem
441 fputs("ERROR: unable to open log file\n", stderrstderr);
442 // failure to open log is serious problem
443 return false;
444 }
445
446 // Load Preferences
447 CPreferences::BuildItemList(thePrefs::GetConfigDir());
448 CPreferences::LoadAllItems( wxConfigBase::Get() );
449
450#ifdef CLIENT_GUI
451 m_skipConnectionDialog = cmdline.Found(wxT("skip")L"skip");
452#else

454 if (cmdline.Found(wxT("use-amuleweb")L"use-amuleweb", &amulewebPath)) {
455 thePrefs::SetWSPath(amulewebPath);
456 AddLogLineNS(CFormat(wxT("Using amuleweb in '%s'.")) % amulewebPath)theLogger.AddLogLine(L"amuleAppCommon.cpp", 456, false, logStandard
, CFormat(L"Using amuleweb in '%s'.") % amulewebPath, true);
457 }
458#endif
459
460 return true;
461}

474bool CamuleAppCommon::CheckPassedLink(const wxString &in, wxString &out, int cat)
475{
476 wxString link(in);
477
478 // restore ASCII-encoded pipes
479 link.Replace(wxT("%7C")L"%7C", wxT("|")L"|");
480 link.Replace(wxT("%7c")L"%7c", wxT("|")L"|");
481
482 if (link.compare(0, 7, wxT("magnet:")L"magnet:") == 0) {
483 link = CMagnetED2KConverter(link);
484 if (link.empty()) {
485 AddLogLineCS(CFormat(wxT("Cannot convert magnet link to eD2k: %s")) % in)theLogger.AddLogLine(L"amuleAppCommon.cpp", 485, true, logStandard
, CFormat(L"Cannot convert magnet link to eD2k: %s") % in, true
);
486 return false;
487 }
488 }
489
490 try {
491 CScopedPtr<CED2KLink> uri(CED2KLink::CreateLinkFromUrl(link));
492 out = uri.get()->GetLink();
493 if (cat && uri.get()->GetKind() == CED2KLink::kFile) {
494 out += CFormat(wxT(":%d")L":%d") % cat;
495 }
496 return true;
497 } catch ( const wxString& err ) {
498 AddLogLineCS(CFormat(wxT("Invalid eD2k link \"%s\" - ERROR: %s")) % link % err)theLogger.AddLogLine(L"amuleAppCommon.cpp", 498, true, logStandard
, CFormat(L"Invalid eD2k link \"%s\" - ERROR: %s") % link % err
, true);
499 }
500 return false;
501}

504/**
505 * Checks permissions on a aMule directory, creating if needed.
506 *
507 * @param desc A description of the directory in question, used for error messages.
508 * @param directory The directory in question.
509 * @param alternative If the dir specified with 'directory' could not be created, try this instead.
510 * @param outDir Returns the used path.
511 * @return False on error.
512 */
513bool CamuleAppCommon::CheckMuleDirectory(const wxString& desc, const CPath& directory, const wxString& alternative, CPath& outDir)
514{
515 wxString msg;
516
517 if (directory.IsDir(CPath::readwritable)) {
518 outDir = directory;
519 return true;
520 } else if (directory.DirExists()) {
521 // Strings are not translated here because translation isn't up yet.
522 msg = CFormat(wxT("Permissions on the %s directory too strict!\n")L"Permissions on the %s directory too strict!\n"
523 wxT("aMule cannot proceed. To fix this, you must set read/write/exec\n")L"aMule cannot proceed. To fix this, you must set read/write/exec\n"
524 wxT("permissions for the folder '%s'")L"permissions for the folder '%s'")
525 % desc % directory;
526 } else if (CPath::MakeDir(directory)) {
527 outDir = directory;
528 return true;
529 } else {
530 msg << CFormat(wxT("Could not create the %s directory at '%s'.")L"Could not create the %s directory at '%s'.")
531 % desc % directory;
532 }
533
534 // Attempt to use fallback directory.
535 const CPath fallback(alternative);
536 if (fallback.IsOk() && (directory != fallback)) {
537 msg << wxT("\nAttempting to use default directory at location \n'")L"\nAttempting to use default directory at location \n'"
538 << alternative << wxT("'.")L"'.";
539 if (theApp->ShowAlert(msg, wxT("Error accessing directory.")L"Error accessing directory.", wxICON_ERROR0x00000200 | wxOK0x00000004 | wxCANCEL0x00000010) == wxCANCEL0x00000010) {
540 outDir = CPath(wxEmptyString);
541 return false;
542 }
543
544 return CheckMuleDirectory(desc, fallback, wxEmptyString, outDir);
545 }
546
547 theApp->ShowAlert(msg, wxT("Fatal error.")L"Fatal error.", wxICON_ERROR0x00000200 | wxOK0x00000004);
548 outDir = CPath(wxEmptyString);
549 return false;
550}
